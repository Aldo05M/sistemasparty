=== PASO 1: CREAR TABLA "participants" ===

Ve a Supabase > SQL Editor y ejecuta:

CREATE TABLE participants (
    id BIGSERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    age INTEGER,
    school TEXT,
    hobby TEXT,
    social TEXT,
    photo_url TEXT,
    votes_count INTEGER DEFAULT 0,
    category TEXT NOT NULL DEFAULT 'rey' CHECK (category IN ('rey', 'reina')),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- category = 'rey' -> usa corona_rey.png
-- category = 'reina' -> usa corona_reina.png

-- Habilitar RLS (Row Level Security)
ALTER TABLE participants ENABLE ROW LEVEL SECURITY;

-- Permitir lectura pública (cualquiera puede ver los participantes)
CREATE POLICY "Lectura pública de participantes"
    ON participants
    FOR SELECT
    USING (true);


=== PASO 2: CREAR TABLA "votes_log" ===

CREATE TABLE votes_log (
    id BIGSERIAL PRIMARY KEY,
    participant_id BIGINT NOT NULL REFERENCES participants(id) ON DELETE CASCADE,
    ip_address TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Habilitar RLS
ALTER TABLE votes_log ENABLE ROW LEVEL SECURITY;

-- Permitir inserción (el RPC se encarga de la validación)
CREATE POLICY "Insertar votos vía RPC"
    ON votes_log
    FOR INSERT
    WITH CHECK (true);

-- Índice para búsquedas rápidas por IP
CREATE INDEX idx_votes_log_ip ON votes_log(ip_address);

-- Índice para búsquedas por participante
CREATE INDEX idx_votes_log_participant ON votes_log(participant_id);


=== PASO 3: CREAR FUNCIÓN RPC PARA VOTAR ===

Esta función valida que la IP no tenga más de 3 votos
y registra el voto de forma atómica (segura).

CREATE OR REPLACE FUNCTION vote_for_participant(
    p_participant_id BIGINT,
    p_ip TEXT
)
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    total_votes INTEGER;
    participant_exists BOOLEAN;
BEGIN
    -- Verificar que el participante existe
    SELECT EXISTS(SELECT 1 FROM participants WHERE id = p_participant_id)
    INTO participant_exists;

    IF NOT participant_exists THEN
        RETURN json_build_object(
            'success', false,
            'message', 'Participante no encontrado'
        );
    END IF;

    -- Contar votos totales de esta IP
    SELECT COUNT(*)
    INTO total_votes
    FROM votes_log
    WHERE ip_address = p_ip;

    -- Verificar límite de 3 votos por IP
    IF total_votes >= 3 THEN
        RETURN json_build_object(
            'success', false,
            'message', 'Ya alcanzaste el límite de 3 votos'
        );
    END IF;

    -- Registrar el voto en votes_log
    INSERT INTO votes_log (participant_id, ip_address)
    VALUES (p_participant_id, p_ip);

    -- Incrementar el contador del participante
    UPDATE participants
    SET votes_count = votes_count + 1
    WHERE id = p_participant_id;

    RETURN json_build_object(
        'success', true,
        'message', 'Voto registrado exitosamente',
        'remaining', 2 - total_votes
    );
END;
$$;

=== PASO 7: CONEXIÓN JAVASCRIPT CON SUPABASE ===

El archivo js/supabase-config.js ya tiene la conexión configurada:

    const SUPABASE_URL = 'https://TU_PROYECTO.supabase.co';
    const SUPABASE_KEY = 'tu_anon_key_aqui';
    const supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);

El archivo js/perreo.js usa supabaseClient para:

1. Leer participantes:
   supabaseClient.from('participants').select('*').order('votes_count', { ascending: false })

2. Votar (RPC):
   supabaseClient.rpc('vote_for_participant', { p_participant_id: 123, p_ip: '1.2.3.4' })

3. Suscripción en tiempo real:
   supabaseClient.channel('perreo-realtime')
       .on('postgres_changes', { event: '*', schema: 'public', table: 'participants' }, callback)
       .subscribe()

=== NOTAS IMPORTANTES ===

- La validación de votos se hace por IP en el backend (función RPC)
  y por localStorage en el frontend (doble capa de seguridad)

- La función RPC usa SECURITY DEFINER para ejecutarse con permisos
  elevados, evitando que el usuario manipule directamente las tablas

- Si necesitas resetear los votos de un usuario para pruebas:
  DELETE FROM votes_log WHERE ip_address = 'IP_DEL_USUARIO';
  
  Y recalcular conteos:
  UPDATE participants p
  SET votes_count = (SELECT COUNT(*) FROM votes_log v WHERE v.participant_id = p.id);

- Para cambiar el límite de votos, modifica:
  * En la función RPC: IF total_votes >= 3
  * En js/perreo.js: const MAX_PERREO_VOTES = 3;
